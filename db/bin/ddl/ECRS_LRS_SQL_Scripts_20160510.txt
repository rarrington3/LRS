-- First create a user ECRS

CREATE USER ECRS WITH ENCRYPTED PASSWORD 'Dev2015ECRS';

-- Alter to add privilege to create db and role for this user
ALTER ROLE ECRS with createdb;
ALTER ROLE ECRS with createrole;

--  Create database ECRSDEV owned by  ECRS 
CREATE DATABASE ecrsdev WITH OWNER ECRS;

--  Select the database 
SELECT current_database();  

--  If the above show anything other than ecrsdev then execute this - Change the Database
\c ecrsdev

-- See the session User
SELECT SESSION_USER, CURRENT_USER; 

-- If the above show anything other than ECRS then execute this – Set the session user as ECRS
set session authorization ECRS;

-- See the session User again to verify the user is ECRS
SELECT SESSION_USER, CURRENT_USER; 

-- Verify the database is ECRSDEV by rerunning
SELECT current_database();  

-- Create user ECRSUSER 
CREATE USER ECRSUSER WITH ENCRYPTED PASSWORD 'Dev2015ECRS';


--  Grant Privilege to create in Database
GRANT CREATE ON DATABASE ecrsdev TO ecrs;
GRANT CREATE ON DATABASE ecrsdev TO ecrsuser;


--  CREATE SCHEMA “ecrs”  in the database created 
CREATE SCHEMA IF NOT EXISTS ecrs  AUTHORIZATION ecrs;

-- GRANT ALL ON SCHEMA ecrs 
GRANT ALL ON SCHEMA ecrs TO ecrs;
GRANT ALL ON SCHEMA ecrs TO ecrsuser;
